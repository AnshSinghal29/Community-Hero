857:     function animateNumber(id, target) {
858:       const el = document.getElementById(id);
859:       const current = parseInt(el.textContent) || 0;
860:       if (current === target) return;
861:       const step = target > current ? 1 : -1;
862:       const steps = Math.min(Math.abs(target - current), 30);
863:       let i = 0;
864:       const timer = setInterval(() => {
865:         el.textContent = Math.round(current + (step * (i / steps) * Math.abs(target - current)));
866:         if (++i > steps) { el.textContent = target; clearInterval(timer); }
867:       }, 20);
868:     }
