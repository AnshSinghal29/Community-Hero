23: ```python
24: # backend/google_drive.py
25: from google.colab import auth
26: from googleapiclient.discovery import build
27: import os
28: 
29: class DriveManager:
30:     def __init__(self):
31:         self.drive = build('drive', 'v3')
32:         self.root_folder_id = self._ensure_root_folder()
33:     
34:     def _ensure_root_folder(self):
35:         """Ensure Community Hero root folder exists in Drive."""
36:         results = self.drive.files().list(
37:             q="name='Community-Hero' and mimeType='application/vnd.google-apps.folder' and trashed=false",
38:             spaces='drive',
39:             pageSize=1
40:         ).execute()
41:         
42:         items = results.get('files', [])
43:         if items:
44:             return items[0]['id']
45:         
46:         # Create if not exists
47:         file_metadata = {
48:             'name': 'Community-Hero',
49:             'mimeType': 'application/vnd.google-apps.folder'
50:         }
51:         folder = self.drive.files().create(body=file_metadata, fields='id').execute()
52:         return folder.get('id')
53:     
54:     def _ensure_category_folder(self, category):
55:         """Ensure category subfolder exists."""
56:         results = self.drive.files().list(
57:             q=f"name='{category}' and mimeType='application/vnd.google-apps.folder' and trashed=false and '{self.root_folder_id}' in parents",
58:             spaces='drive',
59:             pageSize=1
60:         ).execute()
61:         
62:         items = results.get('files', [])
63:         if items:
64:             return items[0]['id']
65:         
66:         # Create if not exists
67:         file_metadata = {
68:             'name': category,
69:             'mimeType': 'application/vnd.google-apps.folder',
70:             'parents': [self.root_folder_id]
71:         }
72:         folder = self.drive.files().create(body=file_metadata, fields='id').execute()
73:         return folder.get('id')
74:     
75:     def upload_issue_image(self, image_bytes, report_id, category):
76:         """Upload issue image to Drive."""
77:         folder_id = self._ensure_category_folder(category)
78:         
79:         file_metadata = {
80:             'name': f'{report_id}.jpg',
81:             'parents': [folder_id]
82:         }
83:         
84:         media = MediaIoBaseUpload(BytesIO(image_bytes), mimetype='image/jpeg', resumable=True)
85:         file = self.drive.files().create(body=file_metadata, media_body=media, fields='id, webViewLink').execute()
86:         
87:         return {
88:             'file_id': file.get('id'),
89:             'web_view_link': file.get('webViewLink'),
90:             'drive_url': f"https://drive.google.com/file/d/{file.get('id')}/view"
91:         }
92: 
93: # backend/app.py - Updated route
94: from google_drive import DriveManager
95: 
96: drive_manager = DriveManager()
97: 
98: @app.route('/api/report', methods=['POST'])
99: def create_report():
100:     """Create report and upload image to Drive."""
101:     
102:     data = request.json
103:     image_file = request.files.get('image')
104:     
105:     # Run agentic analysis
106:     analysis = analyze_issue(...)
107:     
108:     # Upload image to Drive
109:     drive_info = drive_manager.upload_issue_image(
110:         image_bytes=image_file.read(),
111:         report_id=report_id,
112:         category=analysis['category']
113:     )
114:     
115:     # Save to Sheets with Drive link
116:     report_id = save_to_sheets(
117:         analysis=analysis,
118:         image_url=drive_info['drive_url'],  # Use Drive link instead of local
119:         timestamp=datetime.now().isoformat()
120:     )
121:     
122:     return jsonify({
123:         'success': True,
124:         'report_id': report_id,
125:         'image_drive_url': drive_info['drive_url'],
126:         'analysis': analysis
127:     })
128: ```

244: # backend/agent.py - Enhanced
245: def analyze_issue_with_context(image_base64, description, lat, lng, recent_reports):
246:     """Add web search context to agent analysis."""
247:     
248:     # Search for recent news
249:     news_context = search_web(
250:         query=f"Ghaziabad infrastructure {get_sector_from_coords(lat, lng)}"

252:     
253:     # Search for weather
254:     weather = search_web(query=f"Ghaziabad weather {datetime.now().strftime('%Y-%m-%d')}")

256:     # Include in agent prompt

258:     Current context:

476: ### Issue: "Sheets API quota exceeded"
477: **Solution**:
478: ```python
479: # Add caching to reduce API calls
480: from functools import lru_cache
481: import time
482: 
483: @lru_cache(maxsize=1)
484: def get_issues_cached():
485:     """Cache Sheets data for 60 seconds."""
486:     return get_from_sheets()
487: 
488: # Refresh every 60 seconds
489: def get_issues_with_cache():
490:     if time.time() % 60 == 0:
491:         get_issues_cached.cache_clear()
492:     return get_issues_cached()
493: ```
