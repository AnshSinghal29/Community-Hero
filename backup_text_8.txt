518: def get_from_db(category: str = None, priority: str = None, limit: int = 100) -> list:
519:     """Fetch issues from SQLite."""
520:     conn = sqlite3.connect(DB_PATH)
521:     conn.row_factory = sqlite3.Row
522:     cursor = conn.cursor()
523:     
524:     query = "SELECT * FROM issues WHERE 1=1"
525:     params = []
526:     
527:     if category:
528:         query += " AND category = ?"
529:         params.append(category)
530:

532:         query += " AND priority = ?"
533:         params.append(priority)
534:     
535:     query += " ORDER BY timestamp DESC LIMIT ?"
536:     params.append(limit)
537:     
538:     cursor.execute(query, params)
539:     rows = cursor.fetchall()
540:     conn.close()

542:     return [dict(row) for row in rows]

544: def get_recent_reports(latitude: float, longitude: float, days: int = 7, radius_meters: int = 50) -> list:

546:     conn = sqlite3.connect(DB_PATH)
547:     conn.row_factory = sqlite3.Row
548:     cursor = conn.cursor()

550:     # Simple distance check (rough)

553:     cursor.execute('''
554:         SELECT * FROM issues 
555:         WHERE timestamp > ?
556:         AND ABS(latitude - ?) < 0.001
